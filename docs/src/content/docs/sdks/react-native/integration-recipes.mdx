## Sign-in gate

Show a sign-in screen until the user is authenticated, then render the app:

```tsx
import { useThinkworkAuth } from "@thinkwork/react-native-sdk";

export function App() {
  const { status } = useThinkworkAuth();

  if (status === "unknown") return <Splash />;
  if (status !== "signed-in") return <SignInScreen />;
  return <AppShell />;
}

function SignInScreen() {
  const { signInWithGoogle } = useThinkworkAuth();
  return (
    <Button onPress={signInWithGoogle} title="Continue with Google" />
  );
}
```

## Agent picker with persistent selection

The SDK returns agents as data. Persistence is a host concern — this recipe uses AsyncStorage.

```tsx
import * as React from "react";
import AsyncStorage from "@react-native-async-storage/async-storage";
import type { Agent } from "@thinkwork/react-native-sdk";

const KEY = "myapp.selectedAgentId";

export function useSelectedAgent(agents: Agent[]) {
  const effective = React.useMemo(
    () => agents.filter((a) => a.type !== "local"),
    [agents],
  );

  const [selectedId, setSelectedId] = React.useState<string | null>(null);
  const [hydrated, setHydrated] = React.useState(false);

  React.useEffect(() => {
    AsyncStorage.getItem(KEY).then((v) => {
      setSelectedId(v ?? null);
      setHydrated(true);
    });
  }, []);

  React.useEffect(() => {
    if (!hydrated || selectedId || effective.length === 0) return;
    const fallback = effective.find((a) => a.role === "team") ?? effective[0];
    setSelectedId(fallback?.id ?? null);
  }, [hydrated, selectedId, effective]);

  const setId = React.useCallback((id: string | null) => {
    setSelectedId(id);
    if (id) AsyncStorage.setItem(KEY, id).catch(() => {});
    else AsyncStorage.removeItem(KEY).catch(() => {});
  }, []);

  const selected = React.useMemo(
    () => effective.find((a) => a.id === selectedId) ?? null,
    [effective, selectedId],
  );

  return {
    agents: effective,
    selectedAgent: selected,
    selectedAgentId: selectedId,
    setSelectedAgentId: setId,
  };
}
```

Use it:

```tsx
const { user } = useThinkworkAuth();
const { agents } = useAgents({ tenantId: user?.tenantId });
const { selectedAgent, setSelectedAgentId } = useSelectedAgent(agents);
```

## Chat screen

The complete chat surface — header, inverted message list, composer — as shipped in the SDK's `examples/chat-ui/ThinkworkChat.tsx`:

```tsx
import { useMessages, useSendMessage, useThread } from "@thinkwork/react-native-sdk";

export function ChatScreen({ threadId, onBack }: { threadId: string; onBack?: () => void }) {
  const { thread } = useThread(threadId);
  const { messages, loading } = useMessages(threadId);
  const sendMessage = useSendMessage();

  const [draft, setDraft] = React.useState("");
  const [sending, setSending] = React.useState(false);

  const submit = async () => {
    const text = draft.trim();
    if (!text || sending) return;
    setDraft("");
    setSending(true);
    try {
      await sendMessage(threadId, text);
    } finally {
      setSending(false);
    }
  };

  // Messages arrive newest-first; reverse for display.
  const timeline = React.useMemo(() => [...messages].reverse(), [messages]);

  return (
    <View>
      <Header title={thread?.title ?? "Chat"} onBack={onBack} />
      {loading && messages.length === 0 ? (
        <ActivityIndicator />
      ) : (
        <FlatList
          data={timeline}
          keyExtractor={(m) => m.id}
          renderItem={({ item }) => (
            <MessageBubble role={item.role} content={item.content ?? ""} />
          )}
        />
      )}
      <Composer value={draft} onChange={setDraft} onSend={submit} disabled={sending} />
    </View>
  );
}
```

The full version with markdown rendering and scroll-to-last-assistant behavior lives at [`examples/chat-ui/ThinkworkChat.tsx`](https://github.com/thinkwork-ai/thinkwork/blob/main/packages/react-native-sdk/examples/chat-ui/ThinkworkChat.tsx) in the repo.

## Atomic thread creation with first message

The dashboard "start a new thread" composer calls a single mutation that creates the thread *and* sends the opening user message in one round-trip:

```tsx
const createThread = useCreateThread();

async function startThread(text: string, agentId: string, tenantId: string) {
  const thread = await createThread({
    tenantId,
    agentId,
    title: text.length > 60 ? text.slice(0, 60) + "…" : text,
    channel: "CHAT",
    firstMessage: text,
  });
  router.push(`/thread/${thread.id}`);
}
```

No post-creation `useSendMessage` call is needed; the server has already dispatched the agent response path.

## Tab-bar unread badge

Drive a tab-bar badge count with the server-side aggregate so it stays O(1) as the thread list grows:

```tsx
import { useAgents, useThinkworkAuth, useUnreadThreadCount } from "@thinkwork/react-native-sdk";

export function BrainTabBadge({ render }: { render: (count: number) => React.ReactNode }) {
  const { user } = useThinkworkAuth();
  const tenantId = user?.tenantId;
  const { agents } = useAgents({ tenantId });
  const { selectedAgent } = useSelectedAgent(agents);
  const { count } = useUnreadThreadCount({
    tenantId,
    agentId: selectedAgent?.id,
  });
  return <>{render(count)}</>;
}
```

Because the unread-count query and the thread-list query share the `Thread` typename in urql's document cache, `useUpdateThread(threadId, { lastReadAt })` decrements the badge *and* updates the list row atomically — no manual refetch wiring.

## Handling "not configured" gracefully

If your host has stages (dev, prod) or embedding contexts where ThinkWork isn't available, gate the provider mount on a host-level check:

```tsx
const isThinkworkConfigured = Boolean(
  process.env.EXPO_PUBLIC_THINKWORK_GRAPHQL_URL &&
  process.env.EXPO_PUBLIC_COGNITO_USER_POOL_ID,
);

return isThinkworkConfigured ? (
  <ThinkworkProvider config={config}>
    <YourApp />
  </ThinkworkProvider>
) : (
  <YourAppWithoutThinkwork />
);
```

Hide tab-bar entries that depend on ThinkWork behind the same check so users never see broken Brain tabs.
